# 2. Definicion de funciones y operaciones aritmeticas
def calcularArea(float base, float altura):
    float area = (base * altura) / 2.0
    return area

float resultadoArea = calcularArea(miBase, miAltura)

Write("El area calculada es:")
Write(resultadoArea)
